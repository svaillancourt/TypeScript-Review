commands

tsc filename.ts 

- this converts your ts to a js file 

tsc -w -p .

- this automatic complies the ts to a js file 
	the -w & -p are flags
- requires a .json file prior

{
    "compilerOptions": {
        "module": "commonjs",
        "noImplicitAny": true,
        "removeComments": true,
        "preserveConstEnums": true,
        "sourceMap": true
    },
    "files": [
        "main.ts"
    ]
}
